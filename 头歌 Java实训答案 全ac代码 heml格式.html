		 * 要求是 百位 十位  各位上的数字不能重复

		 * 比如：122  131  121  232 等都不满足条件

			  array[i] = sc.nextInt();

			//通过第二层循环控制十位的数字    array[j]表示十位的值

    			sb.append(field[i].getName()).append("=?,");

    	//默认第一个属性为主键，切更改时通过第一个属性进行更改

    	String sql = sb.toString()+";";
    	Connection conn = getConnection();//获取连接对象
